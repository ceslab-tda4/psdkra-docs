198 static uint32_t EthRemoteCfg_getPortNum(EthRemoteCfg_VirtPort portId)
199 {
200  uint32_t portNum;
201 
202  if (EthRemoteCfg_isSwitchPort(portId))
203  {
204  portNum = (uint32_t)portId;
205  }
206  else
207  {
208  portNum = (uint32_t)(portId - ETHREMOTECFG_MAC_PORT_1) + 1U;
209  }
210 
211  return portNum;
212 }

228 static Enet_MacPort EthRemoteCfg_getMacPort(EthRemoteCfg_VirtPort portId)
229 {
230  Enet_MacPort macPort = ENET_MAC_PORT_INV;
231 
232  if (EthRemoteCfg_isMacPort(portId))
233  {
234  macPort = ENET_MACPORT_DENORM(portId - ETHREMOTECFG_MAC_PORT_1);
235  }
236 
237  return macPort;
238 }
